19 class LogConcaveFunction {

28  virtual double operator()(double x) = 0;

30  return LEFT_TRANFORMED_BOUND;

32  double getMaxRange() {
33  return RIGHT_TRANFORMED_BOUND;
34  }

36  return _logEvaluate(_min_log_range);

39  return _logEvaluate(_max_log_range);

41  double _transformInput(double input) {
42 
43  return LEFT_TRANFORMED_BOUND
44  + ((input - _min_log_range) / (_max_log_range - _min_log_range))
45  * (RIGHT_TRANFORMED_BOUND - LEFT_TRANFORMED_BOUND);
46  }
47 
48  double _inverseTransformInput(double input) {
49 
50  return _min_log_range
51  + ((input - LEFT_TRANFORMED_BOUND)
52  / (RIGHT_TRANFORMED_BOUND - LEFT_TRANFORMED_BOUND))
53  * (_max_log_range - _min_log_range);
54  }
55  bool isPointMass() {
56  return (exp(_max_log_range) - exp(_min_log_range))
57  / (0.5 * (exp(_max_log_range) + exp(_min_log_range))) <= TOL;
58  }

61  virtual double _logEvaluate(double x) const = 0;
62  void _setRange(double feasible_pt, double min_range_tmp,
63  double max_range_tmp) {
64  _max_computed_log_pt = feasible_pt;
65  _min_log_range = min_range_tmp;
66  _max_log_range = max_range_tmp;
67  do {
68  _scale = 0.0;
69  _setBounds(_max_computed_log_pt, _min_log_range, _max_log_range);
70  _setScale();
71  } while (!_range_properly_set());
72  }
73  bool _range_properly_set() {
74  return _min_bound_val >= _max_computed_log_val + MIN_BOUND_VAL;
75  }

77  void _setBounds(double feasible_pt, double min_range_tmp,
78  double max_range_tmp) {
79  double feasible_val = _logEvaluate(feasible_pt);
80  if (feasible_val <= -DBL_MAX || feasible_val >= DBL_MAX
81  || feasible_val != feasible_val) {
82  double min_val = _logEvaluate(min_range_tmp);
83  double max_val = _logEvaluate(max_range_tmp);
84  std::ostringstream os;
85  os
86  << "Could not sample from log concave function, since at feasible point "
87  << feasible_pt << " it is " << feasible_val << std::endl;
88  throw std::runtime_error(os.str());
89  }
90  _max_computed_log_val = feasible_val;
91  _max_computed_log_pt = feasible_pt;
92 
93  double min_val = _logEvaluate(min_range_tmp);
94  _updateMaxPt(min_val, min_range_tmp);
95  double max_val = _logEvaluate(max_range_tmp);
96  _updateMaxPt(max_val, max_range_tmp);
97 
98  _min_bound_val = feasible_val + MIN_BOUND_VAL;
99  _max_bound_val = feasible_val + MAX_BOUND_VAL;
100 
101  if (min_val < _min_bound_val) {
102  _min_log_range = _findSmallesFeasiblePoint(min_range_tmp,
103  feasible_pt);
104  } else {
105  _min_log_range = min_range_tmp;
106  }
107 
108  if (max_val < _min_bound_val) {
109  _max_log_range = _findSmallesFeasiblePoint(max_range_tmp,
110  feasible_pt);
111  } else {
112  _max_log_range = max_range_tmp;
113  }
114 
115  }

117  double _findSmallesFeasiblePoint(double infeasible_pt, double feasible_pt) {
118 
119  double m_pt = (infeasible_pt + feasible_pt) / 2.0;
120  double m_val = _logEvaluate(m_pt);
121  _updateMaxPt(m_val, m_pt);
122 
123  if (m_val > _min_bound_val) {
124  if (m_val <= _max_bound_val) {
125  return m_pt;
126  } else {
127  return _findSmallesFeasiblePoint(infeasible_pt, m_pt);
128  }
129  } else {
130  return _findSmallesFeasiblePoint(m_pt, feasible_pt);
131  }
132  }

134  void _setScale() {
135  _scale = 0.0;
136  double scale = -DBL_MAX;
137  double new_scale = 0.0;
138  double x = _min_log_range;
139  while (x <= _max_log_range) {
140  new_scale = _logEvaluate(x);
141  if (new_scale > scale) {
142  scale = new_scale;
143  _updateMaxPt(scale, x);
144  }
145  x += (_max_log_range - _min_log_range) / 100;
146  }
147  _scale = scale;
148  }

152  void _updateMaxPt(double new_val, double new_pt) {
153  if (new_val > _max_computed_log_val) {
154  _max_computed_log_val = new_val;
155  _max_computed_log_pt = new_pt;
156  }
157  }
