102  _z.clear();
103  for (std::size_t i = 0; i < _points.size() - 1; i++) {
104  double z = (_points[i + 1].logf() - _points[i].logf()
105  - _points[i + 1].x() * _points[i + 1].dlogf()
106  + _points[i].x() * _points[i].dlogf())
107  / (_points[i].dlogf() - _points[i + 1].dlogf());

112  void _updateDerivatives(double f_min, double f_max, double min,
113  double max) {
114  double log_f_min = log(f_min);
115  double log_f_max = log(f_max);
116  double dlogf_l = (_points[0].logf() - log_f_min)
117  / (_points[0].x() - min);
118  for (std::size_t i = 0; i < _points.size() - 1; i++) {
119  double dlogf_r = (_points[i + 1].logf() - _points[i].logf())
120  / (_points[i + 1].x() - _points[i].x());
121  if (_points[i].dlogf() < dlogf_r || _points[i].dlogf() > dlogf_l) {
122  _points[i].setDLogf((dlogf_l + dlogf_r) / 2.0);
123  }
124  }
125  double dlogf_r = (log_f_max - _points.back().logf())
126  / (max - _points.back().x());
127  if (_points.back().dlogf() < dlogf_r
128  || _points.back().dlogf() > dlogf_l) {
129  _points.back().setDLogf((dlogf_l + dlogf_r) / 2.0);
130  }
131  }
